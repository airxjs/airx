describe('renderToString', () => {
  it('should output stable HTML', async () => {
    const app = createSSRApp(createElement('div', null, 'Hello'))
    const result1 = await app.renderToString()
    const result2 = await app.renderToString()
    expect(result1.html).toBe(result2.html)
  })
})
